      function arrayBuffer2HexString(buffer){
          const  arr = Array.prototype.map.call(new Uint8Array(buffer), function (byte) {
    return ('00' + byte.toString(16)).slice(-2)
  })

const res = arr.join('')
return res
}

const buffer = new ArrayBuffer(4)
console.log(buffer)

const view = new DataView(buffer)

view.setInt8(0,0x12)
view.setInt8(1,0x34)
view.setInt8(2,0x56)
view.setInt8(3,0x78)
